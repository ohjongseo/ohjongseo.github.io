# 선형대수(Numpy 실습)

import numpy as np

### 행렬과 벡터의 코딩 및 연산

# 행렬 코딩
A = np.array([[3, 1, 1], [1, -2, -1], [1, 1, 1]])
print(A)
print(np.shape(A), A.shape) # A의 차원을 확인하는 2가지 방법

# 벡터 코딩
b = np.array([4, 1, 2])
print(b)
print(np.shape(b), b.shape)

# 역행렬 구하기
A_inv = np.linalg.inv(A)
print(A_inv)
print(np.shape(A_inv))

# 역행렬을 이용한 선형시스템 Ax=b의 해 구하기
# x = np.matmul(A_inv, b)
x = A_inv @ b
print(x)
print(np.shape(x))

## 결과검증
## bb = np.matmul(A, x)
bb = A @ x

print(np.shape(bb))
print(bb)

if np.linalg.norm(b - bb) < 1e-3:
    print("OK")
else:
    print("something wrong")
